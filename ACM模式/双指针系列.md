

# [AMC替换数字]([题目页面 (kamacoder.com)](https://kamacoder.com/problempage.php?pid=1064))

```C++
#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;

int main(){
    string s;
    int count = 0;
    int sOldSize = 0;
    int sNewSize = 0;
    
    while (cin >> s){
        for (char i : s){
            if (i >= '0' && i <= '9') count++;
        }
    }
    
    sOldSize = s.size();
    s.resize(sOldSize + 5 * count);
    sNewSize = s.size();
    
    for (int i = sNewSize, j = sOldSize; i > j; i--, j--){
        if (s[j] > '9' || s[j] < '0') s[i] = s[j];
        else {
            s[i] = 'r';
            s[i - 1] = 'e';
            s[i - 2] = 'b';
            s[i - 3] = 'm';
            s[i - 4] = 'u';
            s[i - 5] = 'n';
            i -= 5;
        } 
    }
    cout << s << endl;
    return 0;
}
```
